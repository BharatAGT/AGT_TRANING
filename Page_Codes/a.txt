page 50101 "Student Line Subpage"
{
    PageType = ListPart;
    SourceTable = "Student Line";
    ApplicationArea = All;

    layout
    {
        area(Content)
        {
            repeater(Group)
            {
                field("Document No."; Rec."Document No.")
                {
                }

                field("Item No."; Rec."Item No.")
                {
                }

                field(Quantity; Rec.Quantity)
                {
                    trigger OnValidate()
                    begin
                        if Rec.Quantity <= 0 then
                            Error('Quantity cannot be zero');
                    end;
                }
            }
        }
    }
}
