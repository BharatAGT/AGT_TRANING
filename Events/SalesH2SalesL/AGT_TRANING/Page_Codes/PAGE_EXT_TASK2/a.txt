pageextension 60110 ItemExtBR extends "Item Card"
{
    layout
    {
        addlast(General)
        {
            field("Lookup Item No."; LookupItemNo)
            {
                ApplicationArea = All;

                trigger OnValidate()
                var
                    ItemRec: Record Item;
                begin
                    if ItemRec.Get(LookupItemNo) then
                        Message('Item Description: %1', ItemRec.Description)
                    else
                        Message('Item not found');
                end;
            }
        }
    }

    var
        LookupItemNo: Code[20];
}


pageextension 60111 CustomerExtBR extends "Customer Card"
{
    layout
    {
        addlast(General)
        {
            field("Search Country"; SearchCountry)
            {
                ApplicationArea = All;

                trigger OnValidate()
                var
                    Cust: Record Customer;
                begin
                    Cust.SetRange("Country/Region Code", SearchCountry);

                    if Cust.FindFirst() then
                        Message('First Customer: %1', Cust.Name)
                    else
                        Message('No customer found');
                end;
            }
        }
    }

    var
        SearchCountry: Code[10];
}

codeunit 60112 VendorFilterBR
{
    trigger OnRun()
    var
        Vend: Record Vendor;
    begin
        Vend.SetFilter(Balance, '>0');

        if Vend.FindFirst() then
            Message('Vendor with Balance > 0 : %1', Vend.Name)
        else
            Message('No Vendor Found');
    end;
}

codeunit 60113 ItemFindLastBR
{
    trigger OnRun()
    var
        ItemRec: Record Item;
    begin
        if ItemRec.FindLast() then
            Message('Last Item: %1 - %2', ItemRec."No.", ItemRec.Description);
    end;
}
